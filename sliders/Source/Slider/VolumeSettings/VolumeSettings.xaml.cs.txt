private void volumeSlider_ValueChanged(object sender, SliderValueChangedEventArgs e)
{
    string glyph;
    if (e.NewValue >= 66)
    {
        glyph = "\uE995";
    }
    else if (e.NewValue >= 33)
    {
        glyph = "\uE994";
    }
    else if (e.NewValue >= 1)
    {
        glyph = "\uE993";
    }
    else
    {
        glyph = "\uE992";
    }

    if (this.volumeFontIcon.Glyph != glyph)
    {
        this.volumeFontIcon.Glyph = glyph;
    }
}
